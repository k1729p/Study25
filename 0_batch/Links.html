GraphiQL

  • Application 'company' http://localhost:9091/graphiql?path=/graphql

Copy mutation or query into the 'GraphiQL' and run this in it: 

mutation Departments {
    mutateDepartments(departments : {
      id: 1,
      name: "D-N-1",
      employees: [{
        id: 11,
        firstName : "E-F-N-11",
        lastName : "E-L-N-11"
        title : DEVELOPER
      }]
    }) {
        success
    }
}

query Departments {
  queryDepartments {
    ... queryDepartment
  }
}
fragment queryDepartment on Department {
    name
    employees {
      ... queryEmployee
    }
}
fragment queryEmployee on Employee {
      lastName
}
